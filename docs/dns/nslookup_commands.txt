nslookup example.com

nslookup -type=MX example.com
# 2 Show mail exchange (MX) records for email servers

nslookup -type=NS example.com
# 3 Show authoritative name servers of the domain

nslookup -type=TXT example.com
# 4 Show TXT records (SPF, DKIM, domain verification info)

nslookup -type=AAAA example.com
# 5 Get IPv6 address (AAAA record)

nslookup -type=CNAME www.example.com
# 6 Show canonical name (CNAME) mapping

nslookup example.com 8.8.8.8
# 7 Query specific DNS server (here Google DNS)

nslookup -type=SOA example.com
# 8 Show Start of Authority (SOA) record (primary DNS + admin info)

nslookup -type=ANY example.com
# 9 Show all available DNS records for the domain
